# lab2 表驱动 scanner 的外置 DFA 定义文件
# 三段：CHARCLASS（字符 → 类） / TRANS（转移） / ACCEPT（接受态 → token） / KEYWORD（关键字）

# === 字符分类 ===
CHARCLASS L     LETTER
CHARCLASS D     DIGIT
CHARCLASS E     EXP
CHARCLASS DOT   .
CHARCLASS PLUS  +
CHARCLASS MIN   -
CHARCLASS STAR  *
CHARCLASS SLA   /
CHARCLASS LT    <
CHARCLASS GT    >
CHARCLASS ASG   =
CHARCLASS BANG  !
CHARCLASS AMP   &
CHARCLASS PIPE  |
CHARCLASS LPAR  (
CHARCLASS RPAR  )
CHARCLASS LBK   [
CHARCLASS RBK   ]
CHARCLASS LBR   {
CHARCLASS RBR   }
CHARCLASS CMA   ,
CHARCLASS COL   :
CHARCLASS SCO   ;

# === 起始 ===
START S0

# === 转移：S0 出发 ===
TRANS S0 L     S_ID
TRANS S0 E     S_ID
TRANS S0 D     S_NUM
TRANS S0 DOT   S_DOT
TRANS S0 PLUS  S_PLUS
TRANS S0 MIN   S_MINUS
TRANS S0 STAR  S_MUL
TRANS S0 SLA   S_DIV
TRANS S0 LT    S_LT
TRANS S0 GT    S_GT
TRANS S0 ASG   S_ASG
TRANS S0 BANG  S_BANG
TRANS S0 AMP   S_AMP1
TRANS S0 PIPE  S_PIPE1
TRANS S0 LPAR  S_LPAR
TRANS S0 RPAR  S_RPAR
TRANS S0 LBK   S_LBK
TRANS S0 RBK   S_RBK
TRANS S0 LBR   S_LBR
TRANS S0 RBR   S_RBR
TRANS S0 CMA   S_CMA
TRANS S0 COL   S_COL
TRANS S0 SCO   S_SCO

# === 标识符 ===
TRANS S_ID L S_ID
TRANS S_ID E S_ID
TRANS S_ID D S_ID

# === 数字 / 浮点 ===
TRANS S_NUM    D   S_NUM
TRANS S_NUM    DOT S_NUMDOT
TRANS S_NUM    E   S_EXP_NEED
TRANS S_NUMDOT D   S_NUMDOT
TRANS S_NUMDOT E   S_EXP_NEED
TRANS S_DOT    D   S_DOTFRAC
TRANS S_DOTFRAC D  S_DOTFRAC
TRANS S_DOTFRAC E  S_EXP_NEED
TRANS S_EXP_NEED   PLUS S_EXP_SIGN
TRANS S_EXP_NEED   MIN  S_EXP_SIGN
TRANS S_EXP_NEED   D    S_EXP_DIGITS
TRANS S_EXP_SIGN   D    S_EXP_DIGITS
TRANS S_EXP_DIGITS D    S_EXP_DIGITS

# === 复合运算符 ===
TRANS S_PLUS PLUS S_PLUSPLUS
TRANS S_PLUS ASG  S_PLUSEQ
TRANS S_LT   ASG  S_LE
TRANS S_GT   ASG  S_GE
TRANS S_ASG  ASG  S_EQ
TRANS S_BANG ASG  S_NE
TRANS S_AMP1 AMP  S_AND
TRANS S_PIPE1 PIPE S_OR

# === 接受状态 → token ===
ACCEPT S_ID         ID
ACCEPT S_NUM        NUM
ACCEPT S_NUMDOT     FLOAT
ACCEPT S_DOTFRAC    FLOAT
ACCEPT S_EXP_DIGITS FLOAT
ACCEPT S_DOT        DOT
ACCEPT S_PLUS       ADD
ACCEPT S_PLUSPLUS   AAA
ACCEPT S_PLUSEQ     AAS
ACCEPT S_MINUS      SUB
ACCEPT S_MUL        MUL
ACCEPT S_DIV        DIV
ACCEPT S_LT         LT
ACCEPT S_LE         LE
ACCEPT S_GT         GT
ACCEPT S_GE         GE
ACCEPT S_ASG        ASG
ACCEPT S_EQ         EQ
ACCEPT S_BANG       NOT
ACCEPT S_NE         NE
ACCEPT S_AND        AND
ACCEPT S_OR         OR
ACCEPT S_LPAR       LPAR
ACCEPT S_RPAR       RPAR
ACCEPT S_LBK        LBK
ACCEPT S_RBK        RBK
ACCEPT S_LBR        LBR
ACCEPT S_RBR        RBR
ACCEPT S_CMA        CMA
ACCEPT S_COL        COL
ACCEPT S_SCO        SCO

# === 关键字（在 ID 中再分流） ===
KEYWORD int     INT
KEYWORD float   FLOAT
KEYWORD void    VOID
KEYWORD if      IF
KEYWORD else    ELSE
KEYWORD while   WHILE
KEYWORD return  RETURN
KEYWORD input   INPUT
KEYWORD print   PRINT
